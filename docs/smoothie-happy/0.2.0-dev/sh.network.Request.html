sh.network. Request

new sh.network.Request(settings)

smoothie-happy.js, line 257

XMLHttpRequest wrapper with Promise logic.

Name Type Description
settings Object

Request settings.

Name Type Default Description
url String

URL with protocol.
method String 'GET' optional

'GET', 'POST', 'DELETE', ...
data Mixed null optional

Data to send with the request.
headers Object null optional

Headers to send with the request.
timeout Integer 5000 optional

Timeout for this request in milliseconds.
xhr Object null optional

An XMLHttpRequest instance or an collection of XMLHttpRequest properties/methods to overwrite.
See:
  • Please read this and that to learn more about promises.
Examples

Single request

sh.network.Request({
    url: 'index.html'
})
.onProgress(function(event) {
    // notify progression
    console.info(event.request._url, '>> progress >>',  event.percent, '%');
})
.then(function(event) {
    // index.html is loaded
    console.info(event.request._url, '>> loaded >>', event.response);

    // return result for final then
    return { event: event, error: false };
})
.catch(function(event) {
    // error loading index.html
    console.warn(event.request._url, '>> error >>', event.response);

    // return error for final then
    return { event: event, error: true };
})
.then(function(result) {
    // finaly ...
    var event    = result.event;
    var logType  = result.error ? 'error' : 'info';
    var logLabel = result.error ? 'error' : 'loaded';

    console[logType](event.request._url, '>>', logLabel, '>>', event.response);
});

Chaining requests

sh.network.Request({
    url: 'index.html'
})
.onProgress(function(event) {
    // notify progression
    console.info(event.request._url, '>> progress >>',  event.percent, '%');
})
.then(function(event) {
    // index.html is loaded
    console.info(event.request._url, '>> loaded >>', event.response);

    // return another request
    return sh.network.Request({
        url: 'not_found.html'
    })
    .onProgress(function(event) {
        // notify progression
        console.info(event.request._url, '>> progress >>',  event.percent, '%');
    });
})
.then(function(event) {
    // not_found.html is loaded
    console.info(event.request._url, '>> loaded >>', event.response);

    // return result for final then
    return { event: event, error: false };
})
.catch(function(event) {
    // error loading index.html or not_found.html
    console.warn(event.request._url, '>> error >>', event.response);

    // return error for final then
    return { event: event, error: true };
})
.then(function(result) {
    // finaly ...
    var event    = result.event;
    var logType  = result.error ? 'error' : 'info';
    var logLabel = result.error ? 'error' : 'loaded';

    console[logType](event.request._url, '>>', logLabel, '>>', event.response);
});
